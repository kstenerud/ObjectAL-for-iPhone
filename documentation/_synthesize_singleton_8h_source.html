00036 /* Synthesize Singleton For Class
00037  *
00038  * Creates a singleton interface for the specified class with the following methods:
00039  *
00040  * + (MyClass*) sharedInstance;
00041  * + (void) purgeSharedInstance;
00042  *
00043  * Calling sharedInstance will instantiate the class and swizzle some methods to ensure
00044  * that only a single instance ever exists.
00045  * Calling purgeSharedInstance will destroy the shared instance and return the swizzled
00046  * methods to their former selves.
00047  *
00048  * 
00049  * Usage:
00050  *
00051  * MyClass.h:
00052  * ========================================
00053  *  #import "SynthesizeSingleton.h"
00054  *
00055  *  @interface MyClass: SomeSuperclass
00056  *  {
00057  *      ...
00058  *  }
00059  *  SYNTHESIZE_SINGLETON_FOR_CLASS_HEADER(MyClass);
00060  *
00061  *  @end
00062  * ========================================
00063  *
00064  *
00065  *  MyClass.m:
00066  * ========================================
00067  *  #import "MyClass.h"
00068  *
00069  *  // This line is optional. Use it if you've enabled GCC_WARN_UNDECLARED_SELECTOR
00070  *  SYNTHESIZE_SINGLETON_FOR_CLASS_PROTOTYPE(MyClass);
00071  *
00072  *  @implementation MyClass
00073  *
00074  *  SYNTHESIZE_SINGLETON_FOR_CLASS(MyClass);
00075  *
00076  *  ...
00077  *
00078  *  @end
00079  * ========================================
00080  *
00081  *
00082  * Note: Calling alloc manually will also initialize the singleton, so you
00083  * can call a more complex init routine to initialize the singleton like so:
00084  *
00085  * [[MyClass alloc] initWithParam:firstParam secondParam:secondParam];
00086  *
00087  * Just be sure to make such a call BEFORE you call "sharedInstance" in
00088  * your program.
00089  */
00090

00093 + (SS_CLASSNAME*) sharedInstance;   \
00094 + (void) purgeSharedInstance;

00306 /* A lesser singleton has a shared instance, but can also be instantiated on its own.
00307  *
00308  * For a lesser singleton, you still use SYNTHESIZE_SINGLETON_FOR_CLASS_HEADER(),
00309  * but use SYNTHESIZE_LESSER_SINGLETON_FOR_CLASS() in the implementation file.
00310  * You must specify which creation methods are to initialize the shared instance
00311  * (besides "sharedInstance") via CALL_LESSER_SINGLETON_INIT_METHOD()
00312  *
00313  * Example:
00314  *
00315  * MyClass.h:
00316  * ========================================
00317  *  #import "SynthesizeSingleton.h"
00318  *
00319  *  @interface MyClass: SomeSuperclass
00320  *  {
00321  *      int value;
00322  *      ...
00323  *  }
00324  *  SYNTHESIZE_SINGLETON_FOR_CLASS_HEADER(MyClass);
00325  *
00326  *  + (void) initSharedInstanceWithValue:(int) value;
00327  *
00328  * - (id) initWithValue:(int) value;
00329  *
00330  *  @end
00331  * ========================================
00332  *
00333  *
00334  *  MyClass.m:
00335  * ========================================
00336  *  #import "MyClass.h"
00337  *
00338  *  // This line is optional. Use it if you've enabled GCC_WARN_UNDECLARED_SELECTOR
00339  *  SYNTHESIZE_SINGLETON_FOR_CLASS_PROTOTYPE(MyClass);
00340  *
00341  *  @implementation MyClass
00342  *
00343  *  SYNTHESIZE_LESSER_SINGLETON_FOR_CLASS(MyClass);
00344  *
00345  *  + (void) initSharedInstanceWithValue:(int) value
00346  *  {
00347  *      CALL_LESSER_SINGLETON_INIT_METHOD(MyClass, initWithValue:value);
00348  *  }
00349  *
00350  *  ...
00351  *
00352  *  @end
00353  * ========================================
00354  *
00355  *
00356  * Note: CALL_LESSER_SINGLETON_INIT_METHOD() will not work if your
00357  * init call contains commas. If you need commas (such as for varargs),
00358  * or other more complex initialization, use the PRE and POST macros:
00359  *
00360  *  + (void) initSharedInstanceComplex

00362  *      CALL_LESSER_SINGLETON_INIT_METHOD_PRE(MyClass);
